        // Show/Hide Sections
        function showSection(section) {
            // Hide all sections
            document.getElementById('home-section').style.display = 'none';
            document.querySelector('.ticker').style.display = 'none';
            document.querySelector('.features').style.display = 'none';
            document.querySelector('.footer').style.display = 'none';
            document.getElementById('dashboard').style.display = 'none';
            document.getElementById('admin-panel').style.display = 'none';
            
            if (section === 'home') {
                document.getElementById('home-section').style.display = 'grid';
                document.querySelector('.ticker').style.display = 'block';
                document.querySelector('.features').style.display = 'block';
                document.querySelector('.footer').style.display = 'block';
            } else if (section === 'dashboard') {
                document.getElementById('dashboard').style.display = 'block';
                initChart();
            } else if (section === 'admin') {
                document.getElementById('admin-panel').style.display = 'block';
            }
        }

        // Demo Login
        function loginDemo() {
            showToast('Welcome Back', 'Successfully logged in as John Doe');
            setTimeout(() => showSection('dashboard'), 1000);
        }

        // Chart Initialization
        let chartInstance;
        function initChart() {
            if (chartInstance) return;
            
            const ctx = document.getElementById('tradingChart').getContext('2d');
            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: ['10:00', '10:05', '10:10', '10:15', '10:20', '10:25', '10:30'],
                    datasets: [{
                        label: 'EUR/USD',
                        data: [1.0840, 1.0842, 1.0845, 1.0843, 1.0848, 1.0850, 1.0845],
                        borderColor: '#64ffda',
                        backgroundColor: 'rgba(100, 255, 218, 0.1)',
                        borderWidth: 2,
                        fill: true,
                        tension: 0.4
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            labels: { color: '#e6f1ff' }
                        }
                    },
                    scales: {
                        x: {
                            ticks: { color: '#8892b0' },
                            grid: { color: 'rgba(100, 255, 218, 0.1)' }
                        },
                        y: {
                            ticks: { color: '#8892b0' },
                            grid: { color: 'rgba(100, 255, 218, 0.1)' }
                        }
                    }
                }
            });
        }

        // Trading Functions
        function switchTab(btn, type) {
            document.querySelectorAll('.trade-tab').forEach(tab => tab.classList.remove('active'));
            btn.classList.add('active');
        }

        function placeTrade() {
            showToast('Trade Executed', 'Your order has been placed successfully');
        }

        // Withdrawal Modal
        function openWithdrawalModal() {
            document.getElementById('withdrawal-modal').style.display = 'flex';
        }

        function closeWithdrawalModal() {
            document.getElementById('withdrawal-modal').style.display = 'none';
        }

        function selectBank(element) {
            document.querySelectorAll('.bank-option').forEach(opt => opt.classList.remove('selected'));
            element.classList.add('selected');
        }

        function processWithdrawal() {
            const amount = document.getElementById('withdraw-amount').value;
            if (!amount || amount <= 0) {
                showToast('Error', 'Please enter a valid amount');
                return;
            }
            showToast('Withdrawal Initiated', `Your withdrawal of $${amount} is being processed`);
            closeWithdrawalModal();
        }

        // Admin Functions
        function showAdminTab(tab) {
            document.querySelectorAll('.admin-menu a').forEach(link => link.classList.remove('active'));
            event.target.classList.add('active');
            
            document.getElementById('admin-users').classList.add('hidden');
            document.getElementById('admin-funding').classList.add('hidden');
            
            if (tab === 'users') document.getElementById('admin-users').classList.remove('hidden');
            if (tab === 'funding') document.getElementById('admin-funding').classList.remove('hidden');
        }

        function openFundModal(username) {
            showAdminTab('funding');
            const select = document.getElementById('fund-user-select');
            for (let i = 0; i < select.options.length; i++) {
                if (select.options[i].text.includes(username)) {
                    select.selectedIndex = i;
                    break;
                }
            }
        }

        function processFunding() {
            const amount = document.getElementById('fund-amount').value;
            const user = document.getElementById('fund-user-select').value;
            if (!amount) {
                showToast('Error', 'Please enter an amount');
                return;
            }
            showToast('Account Funded', `Successfully added $${amount} to ${user.split('(')[0]}`);
            document.getElementById('fund-amount').value = '';
        }

        // Chat Functions
        function toggleChat() {
            document.getElementById('chat-window').classList.toggle('active');
        }

        function handleChatKey(e) {
            if (e.key === 'Enter') sendMessage();
        }

        function sendMessage() {
            const input = document.getElementById('chat-input');
            const message = input.value.trim();
            if (!message) return;
            
            const chatBody = document.getElementById('chat-body');
            
            // User message
            const userMsg = document.createElement('div');
            userMsg.className = 'chat-message user';
            userMsg.textContent = message;
            chatBody.appendChild(userMsg);
            
            input.value = '';
            chatBody.scrollTop = chatBody.scrollHeight;
            
            // Auto reply
            setTimeout(() => {
                const supportMsg = document.createElement('div');
                supportMsg.className = 'chat-message support';
                supportMsg.textContent = "Thank you for your message. A support agent will assist you shortly. For immediate help, please use WhatsApp.";
                chatBody.appendChild(supportMsg);
                chatBody.scrollTop = chatBody.scrollHeight;
            }, 1000);
        }

        // Toast Notification
        function showToast(title, message) {
            const toast = document.getElementById('toast');
            document.getElementById('toast-title').textContent = title;
            document.getElementById('toast-message').textContent = message;
            toast.classList.add('show');
            setTimeout(() => toast.classList.remove('show'), 3000);
        }

        // Close modal on outside click
        window.onclick = function(event) {
            const modal = document.getElementById('withdrawal-modal');
            if (event.target === modal) {
                closeWithdrawalModal();
            }
        }
